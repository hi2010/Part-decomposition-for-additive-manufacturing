51  template <typename T>
52  T parsType(std::string option, T defaultVal) {
53  std::string strVal = getCmdOption(option);
54  if (!strVal.empty()) {
55  if (std::is_same<T, double>::value) return stod(strVal);
56  else if (std::is_same<T, long>::value) return stol(strVal);
57  else if (std::is_same<T, unsigned int>::value) return static_cast<unsigned int>(stoul(strVal));
58  else if (std::is_same<T, int>::value) return stoi(strVal);
59  }
60  return defaultVal;
61  }
