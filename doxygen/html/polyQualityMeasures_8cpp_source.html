43  double roughness(MeshWithAttributes* meshwa) {
44  // the 4 here is kinda rand. If on assumes approx .5 min error per calculation this means upto 8 erroneous calcuations
45  // but this should be taken with a grain of salt as there could be worse cases so... jus a rndm assumption
46  auto errRange = std::numeric_limits<double>::min()*4;
47  auto upperErrRange = 1 - errRange;
48  auto buildDir = meshwa->printDir;
49  buildDir.Normalize();
50 
51  // iterate over the faces of the component to calculate the per face cost and add it up as component cost
52  double totalCost = 0.0;
53  auto cellNorms = meshwa->getMeshCellNormals();
54  auto cellAreas = meshwa->getMeshTriAreas();
55  auto nNorms = cellNorms->GetNumberOfTuples();
56  for (vtkIdType i = 0; i < nNorms; i++) {
57  auto vNorm = vtkVector3d(cellNorms->GetTuple(i));
58  // vNorm.normalize(); should not be used as i am calculating normalized normals using vtkPolyDataNormals for calculation
59  // the cost function from the paper says the angle should lie in between (0, pi). For cosAngle this is (-1,1), otherwise use 0
60  // this is stange cause the angle is inbetween [0, pi] which would make the value at 0 and pi == 0 and jump in the direct
61  // vincinity to 1. On the other hand this seems legit, cause for angle = 0 and pi this means the face is horizontal
62  // -> any small deviation means the face is very flat -> hich staristep effect
63  // for reasons of calculation errors, ... the absolute 0 and pi should be converted to a small range
64  // this converts to: if cosAngle in [-e,e] or in [1-e, 1+e] := 0
65  auto diffAngle = lVtkHelper::cosAngleNormVectors(vNorm, buildDir);
66  // !!! there is a not at the begining
67  // err range is because rounding error might avoid landingn on 0 or pi
68  if (!((diffAngle > errRange) || (diffAngle < upperErrRange))) {
69  // the angle is within the range
70  auto absDiffAngle = fabs(diffAngle);
71  double triAr = cellAreas->GetComponent(i,0);
72  totalCost += absDiffAngle * triAr;
73  }
74  }
75 
76  if (printResults) std::cout << __func__ << " run with points: " << totalCost << std::endl;
77  return totalCost;
78  }

80  double overhangArea(MeshWithAttributes* meshwa) {
81  // assumed: trimesh
82  auto poly = meshwa->mesh;
83  auto nCells = poly->GetNumberOfCells();
84  auto cellAreas = meshwa->getMeshTriAreas();
85  // critical overhang angle in rad
86  double overhangAngle = meshwa->overhangAngle;
87  double criticalCosAngle = cos(overhangAngle);
88  // according to the paper the printing bed normal should be used here, but for xyz the printing bed normal is just the (in this case negative) printing direction (z-Axis)
89  vtkVector3d printDir = lVtkHelper::invertVector(meshwa->printDir);
90 
91  // minimum z height needed to not be considered on bed
92  double zTol = .000001;
93 
94  double ovhArea = 0;
95 
96  auto cellNorms = meshwa->getMeshCellNormals();
97 
98  // #pragma omp parallel for -> seems unsafe
99  for (vtkIdType i = 0; i < nCells; i++) {
100  vtkVector3d cellNorm = vtkVector3d(cellNorms->GetTuple(i));
101  auto curCosAngl = lVtkHelper::cosAngleNormVectors(cellNorm, printDir);
102 
103  // TODO: this is from paper, but really for 0? means ground plate?
104  if (0 <= curCosAngl && curCosAngl <= criticalCosAngle) {
105  // check if ground level / is on bed?
106  vtkNew<vtkGenericCell> cll;
107  poly->GetCell(i, cll);
108  auto pts = cll->GetPoints();
109  bool onBed = true;
110  // is any point of the polygon above the bed level? If yes -> add area to cost
111  for (vtkIdType ptIdx = 0; ptIdx < pts->GetNumberOfPoints(); ptIdx ++) {
112  double pt[3];
113  pts->GetPoint(ptIdx, pt);
114  if (abs(pt[2]) > zTol) {
115  onBed = false;
116  break;
117  }
118  }
119  if (onBed == false) {
120  // this ones
121  auto triAr = cellAreas->GetComponent(i, 0);
122  ovhArea += triAr * fabs(curCosAngl);
123  }
124  } // if the facets normal is not in a critical angle (to flat overhang), nothing to do
125  }
126 
127  if (printResults) std::cout << __func__ << " run with points: " << ovhArea << std::endl;
128  return ovhArea;
129  }

134  double sharpness(MeshWithAttributes* meshwa) {
135 
136  // in a valid mesh at most two facets share a edge -> calulatinng something with respect to every neighbour of a facet
137  // results in 2* the result as if on would only calculate it per edge -> /2 = per edge
138  //auto cIt = polydata->NewCellIterator();
139  //for (cIt->InitTraversal(); !cIt->IsDoneWithTraversal(); cIt->GoToNextCell()) {
140  // upper bound in rad = 90° -> cehck paper for used value this ones the mathematical definition of sharp
141  double sharpCornersAngle = meshwa->sharpCornersAngle;
142  auto poly = meshwa->mesh;
143  auto cellNorms = meshwa->getMeshCellNormals();
144  vtkIdType nCells = poly->GetNumberOfCells();
145  double cost = 0;
146  // needed for the celledgeneighbors
147  meshwa->createLinksIfNeeded();
148  // -1: less than 3 edges, else last occurence of not == 1 neigbor or 1 if 1 neighbor
149  // used for debug and visual, might be removed
150  if (poly->GetCellData()->HasArray("NotOneNeighbor")) poly->GetCellData()->RemoveArray("NotOneNeighbor");
151  vtkNew<vtkIdTypeArray> arNotOneNeighbor;
152  arNotOneNeighbor->SetName("NotOneNeighbor");
153  arNotOneNeighbor->SetNumberOfComponents(1);
154  arNotOneNeighbor->SetNumberOfTuples(poly->GetNumberOfCells());
155  // TODO: most likely safe
156  //#pragma omp parallel for
157  for (vtkIdType i = 0; i < nCells; i++) {
158 
159  auto curNorm = vtkVector3d(cellNorms->GetTuple(i));
160  auto curCell = poly->GetCell(i);
161  auto nEdges = curCell->GetNumberOfEdges();
162  if (nEdges < 3) {
163  arNotOneNeighbor->SetComponent(i, 0, -2);
164  continue;
165  }
166  vtkIdType not3NNeighbors = 1;
167  //assert(nEdges == 3); -> no because clipClosed may produce invalid, also may contain lines, points, ...
168  for (vtkIdType k = 0; k < nEdges; k++) {
169  auto curEdge = curCell->GetEdge(k);
170  auto p0 = curEdge->GetPointIds()->GetId(0);
171  auto p1 = curEdge->GetPointIds()->GetId(1);
172 
173  vtkNew<vtkIdList> cellNeighbors;
174  poly->GetCellEdgeNeighbors(i, p0, p1, cellNeighbors);
175 
176  auto nNeigbors = cellNeighbors->GetNumberOfIds();
177  if (nNeigbors != 1) {
178  not3NNeighbors = nNeigbors;
179  }
180 
181  //assert(nNeigbors==1);
182  auto nSharpNeig = 0;
183  auto tempCost = 0;
184 
185  for (vtkIdType j = 0; j < nNeigbors; j++) {
186  auto neighborId = cellNeighbors->GetId(j);
187  auto neigNorm = vtkVector3d(cellNorms->GetTuple(neighborId));
188  auto curAngle = lVtkHelper::angleNormVectors(curNorm, neigNorm);
189  if (curAngle >= sharpCornersAngle) {
190  // sharp corner -> calculate the length of the edge and add it to the cost
191  double pc0 [3];
192  double pc1 [3];
193  // not threadsafe it says
194  poly->GetPoint(p0, pc0);
195  poly->GetPoint(p1, pc1);
196  double subVal[3];
197  vtkMath::Subtract(pc0, pc1, subVal);
198  auto edgeLen = vtkMath::Norm(subVal);
199  tempCost += edgeLen;
200  nSharpNeig += 1;
201  }
202  }
203  // remove effects that may occur if more than one neighbor exist
204  // sadly this is what clipClosedSurfaceProduces
205  if (nSharpNeig > 0) cost += tempCost / nSharpNeig;
206  }
207  //arNotOneNeighbor->InsertComponent(i, 0, not3NNeighbors);
208  arNotOneNeighbor->SetComponent(i, 0, not3NNeighbors);
209  }
210  // used for debug and visual, might be removed
211  poly->GetCellData()->AddArray(arNotOneNeighbor);
212  // in a well defined tri mesh at most 2 facets share a edge.
213  // As the cost is calulated for all facets for their neighbors every edge is considered twice
214  // Therefore a edge that is sharp is added twice to the cost -> effect. doubling calc. cost.
215  cost /= 2;
216 
217  if (printResults) std::cout << __func__ << " run with points: " << cost << std::endl;
218  return cost;
219  }
