39 bool OCLHelper::GetPlatformsInfo(std::vector<std::string>& info, const std::string& indentation)
40 {
41  const char* platformInfoParameters[] = { "CL_PLATFORM_NAME",
42  "CL_PLATFORM_VENDOR",
43  "CL_PLATFORM_VERSION",
44  "CL_PLATFORM_PROFILE",
45  "CL_PLATFORM_EXTENSIONS" };
46 
47  cl_uint numPlatforms;
48  m_lastError = clGetPlatformIDs(1, NULL, &numPlatforms);
49  if (m_lastError != CL_SUCCESS)
50  return false;
51 
52  cl_platform_id* platforms = new cl_platform_id[numPlatforms];
53  m_lastError = clGetPlatformIDs(numPlatforms, platforms, NULL);
54  if (m_lastError != CL_SUCCESS) {
55  delete[] platforms;
56  return false;
57  }
58 
59  size_t bufferSize = 4096;
60  char* buffer = new char[bufferSize];
61  size_t size;
62  info.resize(numPlatforms);
63  for (cl_uint i = 0; i < numPlatforms; ++i) {
64  for (int j = CL_PLATFORM_PROFILE; j <= CL_PLATFORM_EXTENSIONS; ++j) {
65  info[i] += indentation + platformInfoParameters[j - CL_PLATFORM_PROFILE] + std::string(": ");
66  m_lastError = clGetPlatformInfo(platforms[i], j, 0, NULL, &size);
67  if (m_lastError != CL_SUCCESS) {
68  delete[] buffer;
69  delete[] platforms;
70  return false;
71  }
72  if (bufferSize < size) {
73  delete[] buffer;
74  bufferSize = size;
75  buffer = new char[bufferSize];
76  }
77  m_lastError = clGetPlatformInfo(platforms[i], j, size, buffer, NULL);
78  if (m_lastError != CL_SUCCESS) {
79  delete[] buffer;
80  delete[] platforms;
81  return false;
82  }
83  info[i] += buffer + std::string("\n");
84  }
85  }
86  delete[] platforms;
87  delete[] buffer;
88  return true;
89 }
